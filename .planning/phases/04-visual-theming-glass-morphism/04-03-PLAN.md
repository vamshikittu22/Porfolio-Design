```css
/**
 * Glass Morphism Utilities
 * 
 * Reusable glass effect classes for consistent visual language.
 * Extends patterns from glass-cards.css and chapter.css.
 * 
 * Usage:
 * - .glass-panel: Large content containers (sections, major divisions)
 * - .glass-card: Interactive cards and smaller components
 * - .glass-overlay: Subtle overlays (headers, footers, modals)
 */

/* ============================================================================
   Base Glass Effects
   ============================================================================ */

.glass-panel {
  background: rgba(var(--background-rgb), 0.6);
  border: 1px solid rgba(255, 255, 255, 0.1);
  border-radius: 24px;
  box-shadow: 
    0 8px 32px rgba(0, 0, 0, 0.12),
    0 2px 8px rgba(0, 0, 0, 0.08),
    inset 0 1px 0 rgba(255, 255, 255, 0.1);
}

.glass-card {
  background: rgba(var(--background-rgb), 0.4);
  border: 1px solid rgba(255, 255, 255, 0.15);
  border-radius: 16px;
  box-shadow: 
    0 4px 24px rgba(0, 0, 0, 0.1),
    0 1px 4px rgba(0, 0, 0, 0.06);
  transition: all 0.3s cubic-bezier(0.4, 0, 0.2, 1);
}

.glass-overlay {
  background: rgba(var(--background-rgb), 0.8);
  backdrop-filter: blur(16px);
  -webkit-backdrop-filter: blur(16px);
  border-bottom: 1px solid rgba(255, 255, 255, 0.05);
}

/* Backdrop blur with progressive enhancement */
@supports (backdrop-filter: blur(12px)) {
  .glass-panel {
    backdrop-filter: blur(12px) saturate(1.2);
    -webkit-backdrop-filter: blur(12px) saturate(1.2);
  }
  
  .glass-card {
    backdrop-filter: blur(10px);
    -webkit-backdrop-filter: blur(10px);
  }
}

/* Fallback for browsers without backdrop-filter */
@supports not (backdrop-filter: blur(12px)) {
  .glass-panel {
    background: rgba(var(--background-rgb), 0.92) !important;
  }
  
  .glass-card {
    background: rgba(var(--background-rgb), 0.85) !important;
  }
}

/* ============================================================================
   Interactive States
   ============================================================================ */

.glass-card:hover {
  border-color: rgba(255, 255, 255, 0.25);
  box-shadow: 
    0 8px 32px rgba(0, 0, 0, 0.15),
    0 2px 8px rgba(0, 0, 0, 0.08);
  transform: translateY(-2px);
}

.glass-card:active {
  transform: translateY(0);
  box-shadow: 
    0 2px 16px rgba(0, 0, 0, 0.08),
    0 1px 4px rgba(0, 0, 0, 0.06);
}

/* ============================================================================
   Reduced Motion Support
   ============================================================================ */

@media (prefers-reduced-motion: reduce) {
  .glass-panel,
  .glass-card,
  .glass-overlay {
    backdrop-filter: none;
    -webkit-backdrop-filter: none;
    transition: none;
  }
  
  .glass-card:hover {
    transform: none;
  }
}

/* ============================================================================
   Dark Mode Adjustments
   ============================================================================ */

@media (prefers-color-scheme: dark) {
  .glass-panel {
    border-color: rgba(255, 255, 255, 0.08);
    box-shadow: 
      0 8px 32px rgba(0, 0, 0, 0.4),
      0 2px 8px rgba(0, 0, 0, 0.2),
      inset 0 1px 0 rgba(255, 255, 255, 0.05);
  }
  
  .glass-card {
    border-color: rgba(255, 255, 255, 0.12);
  }
}

/* ============================================================================
   Typography Scale
   ============================================================================ */

.heading-xl {
  font-size: clamp(2.5rem, 5vw, 4rem);
  line-height: 1.1;
  font-weight: 800;
  letter-spacing: -0.02em;
}

.heading-lg {
  font-size: clamp(2rem, 4vw, 3rem);
  line-height: 1.2;
  font-weight: 700;
  letter-spacing: -0.01em;
}

.heading-md {
  font-size: clamp(1.5rem, 3vw, 2rem);
  line-height: 1.3;
  font-weight: 600;
}

.heading-sm {
  font-size: clamp(1.125rem, 2vw, 1.5rem);
  line-height: 1.4;
  font-weight: 600;
}

.body-lg {
  font-size: clamp(1.125rem, 1.5vw, 1.25rem);
  line-height: 1.6;
}

.body-md {
  font-size: 1rem;
  line-height: 1.5;
}

.caption {
  font-size: 0.875rem;
  line-height: 1.4;
  opacity: 0.8;
}
```
